```python
def _build_cluster_input(items, catalog_items=None):
    lines = []
    # 기존 카탈로그 품목 섹션 삽입
    if catalog_items:
        lines.append("## 기존 카탈로그 품목 (representative_name 작성 시 이 이름을 우선 사용)")
        for ci in catalog_items:
            try:
                aliases = json.loads(ci["aliases"] or "[]")
            except Exception:
                aliases = []
            alias_str = ", ".join(aliases[:5]) if aliases else ""
            entry = f"  - {ci['name_canonical']}"
            if alias_str:
                entry += f" (별칭: {alias_str})"
            lines.append(entry)
        lines.append("")
    lines.append(f"## 신규 입찰 품목 ({len(items)}개 / ...)")
    # ... 기존 품목 목록 출력
    return "\n".join(lines)
```

### 3-2. `accept_cluster` 시 중복 `catalog_item` 생성 방지

#### 문제
`accept_cluster` 실행 시 항상 새 `catalog_item` UUID를 생성 → 동일 이름 품목이 여러 개 생성되어 `price_history` 체인 단절.

#### 수정 파일
`qdbt/src/extractors/catalog_clusterer.py`

#### 수정 내용

```python
# ⑤ catalog_items: 동일 이름 기존 항목 재사용 / 없으면 신규 생성
# 1) name_canonical 정확 매칭 (대소문자 무시)
existing_ci = conn.execute("""
    SELECT catalog_item_id, name_canonical, aliases
    FROM catalog_items
    WHERE LOWER(name_canonical) = LOWER(?) AND is_active = 1
    LIMIT 1
""", (final_name,)).fetchone()

# 2) aliases 역방향 매칭 (final_name이 기존 catalog_item의 별칭인 경우)
if not existing_ci:
    for ci_row in conn.execute(
        "SELECT catalog_item_id, name_canonical, aliases FROM catalog_items WHERE is_active=1"
    ).fetchall():
        try:
            al = json.loads(ci_row["aliases"] or "[]")
        except Exception:
            al = []
        if any(a.lower() == final_name.lower() for a in al):
            existing_ci = ci_row
            break

if existing_ci:
    # 기존 재사용 — aliases 병합만
    catalog_item_id = existing_ci["catalog_item_id"]
    prev_aliases = json.loads(existing_ci["aliases"] or "[]")
    merged_aliases = list(set(prev_aliases) | new_aliases_set)
    if existing_ci["name_canonical"].lower() != final_name.lower():
        merged_aliases = list(set(merged_aliases) | {final_name})
    conn.execute(
        "UPDATE catalog_items SET aliases=?, updated_at=? WHERE catalog_item_id=?",
        (json.dumps(merged_aliases, ensure_ascii=False), now, catalog_item_id)
    )
else:
    # 신규 생성
    catalog_item_id = str(uuid.uuid4())
    aliases = list(new_aliases_set)
    conn.execute("""
        INSERT INTO catalog_items (catalog_item_id, category_id, name_canonical,
            aliases, is_active, created_by, created_at, updated_at)
        VALUES (?, ?, ?, ?, 1, ?, ?, ?)
    """, (catalog_item_id, category_id, final_name,
          json.dumps(aliases, ensure_ascii=False), user_id, now, now))
```

```python
# 마우스오버용 업체별 상세 내역 — catalog_item당 최저가 1건만
det_rows = c.execute(f"""
```

## 6. 클러스터 최저가 계산 기준 수정

### 문제
클러스터 내 `min_price` / `min_vendor`를 개별 항목의 `unit_price` 최솟값으로 계산 → 한 업체가 여러 항목을 제출한 경우(예: F글로벌 = UTP 배선 + 광케이블) 잘못된 최저 업체 선정.

```
구내 배선 공사 클러스터:
  C공업:   17,800,000  (1개 항목)
  F글로벌:  9,500,000 + 12,000,000 = 21,500,000  (2개 항목)

수정 전 → min: F글로벌 (9,500,000)  ← 잘못됨
수정 후 → min: C공업   (17,800,000) ← 올바름
```

### 수정 파일
`qdbt/src/db/queries.py`

### 수정 내용

```python
# 수정 전
all_priced = [(ci["vendor_name"], ci["unit_price"])
              for ci in cluster_items if ci["unit_price"]]
min_vendor, min_price = (
    min(all_priced, key=lambda x: x[1]) if all_priced else (None, None)
)

# 수정 후 — 업체별 amount 합계 기준
from collections import defaultdict as _dd
vendor_totals = _dd(float)
for ci in cluster_items:
    vendor_totals[ci["vendor_name"]] += (ci["amount"] or 0)

if vendor_totals:
    min_vendor = min(vendor_totals, key=lambda v: vendor_totals[v])
    min_price  = vendor_totals[min_vendor]
    if len(vendor_totals) > 1:
        max_vendor = max(vendor_totals, key=lambda v: vendor_totals[v])
        max_price  = vendor_totals[max_vendor]
    else:
        max_vendor = max_price = None
else:
    min_vendor = min_price = max_vendor = max_price = None
```
